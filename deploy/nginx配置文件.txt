# CloudDisk Pro Nginx 部署示例（含基础限流防护）

```nginx
# 在 http {} 块中定义限流 zone
limit_req_zone $binary_remote_addr zone=cd_api:10m rate=30r/s;
limit_req_zone $binary_remote_addr zone=cd_login:10m rate=5r/s;
limit_conn_zone $binary_remote_addr zone=cd_conn:10m;

upstream clouddisk_backend {
    server 127.0.0.1:8088;
    keepalive 32;
}

server {
    listen 443 ssl;
    server_name disk.example.com;

    # ssl_certificate ...
    client_max_body_size 20g;
    client_body_timeout 120s;
    client_header_timeout 30s;
    send_timeout 120s;
    proxy_read_timeout 3600s;
    proxy_send_timeout 3600s;
    proxy_connect_timeout 30s;

    # 单 IP 最大并发连接
    limit_conn cd_conn 50;

    location / {
        root /www/clouddisk/dist;
        try_files $uri $uri/ /index.html;
    }

    location /api/auth/login {
        limit_req zone=cd_login burst=10 nodelay;
        proxy_pass http://clouddisk_backend/api/auth/login;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    location /api/auth/register {
        limit_req zone=cd_login burst=5 nodelay;
        proxy_pass http://clouddisk_backend/api/auth/register;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
    }

    location /api/ {
        limit_req zone=cd_api burst=60 nodelay;
        proxy_pass http://clouddisk_backend/api/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        client_max_body_size 20g;
    }

    location /ws/ {
        proxy_pass http://clouddisk_backend/ws/;
        proxy_http_version 1.1;
        proxy_set_header Upgrade $http_upgrade;
        proxy_set_header Connection "upgrade";
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
        proxy_read_timeout 3600s;
    }

    location /share/ {
        limit_req zone=cd_api burst=30 nodelay;
        proxy_pass http://clouddisk_backend/share/;
        proxy_set_header Host $host;
        proxy_set_header X-Real-IP $remote_addr;
    }

    location /onlyoffice/ {
        proxy_pass http://127.0.0.1:8082/;
        proxy_set_header Host $host;
    }

    location /minio-cdn/ {
        proxy_pass http://127.0.0.1:9000/;
        proxy_set_header Host $host;
        proxy_buffering off;
        client_max_body_size 20g;
    }
}
```

注意：

- **必须**由 Nginx 设置 `X-Real-IP`，后端限流/防刷才可信；禁止公网直连 8088
- `limit_req` / `limit_conn` 为第一道防线，应用层 Redis/Sentinel 为第二道
- `client_max_body_size` 需与后端 `clouddisk.upload.max-file-size`（20GB）一致
- WebSocket 上传进度需单独配置 `/ws/` 代理
- `proxy_pass` 末尾斜杠与 location 保持一致，避免 404/405
